## 3. CHAT INTERFACE (Gemini API Example)

```jsx
import React, { useState } from 'react';

export function AIChat() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  const sendMessage = async () => {
    if (!input.trim()) return;
    const userMsg = { role: 'user', content: input };
    setMessages(prev => [...prev, userMsg]);
    setInput('');
    setIsLoading(true);

    try {
      const res = await fetch('/api/chat', {
        method: 'POST',
        body: JSON.stringify({ prompt: input }),
      });
      const data = await res.json();
      setMessages(prev => [...prev, { role: 'ai', content: data.text }]);
    } catch (error) {
      swal.error('AI Error', 'Failed to get response');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="chat-container">
      <div className="chat-messages">
        {messages.map((m, i) => (
          <div key={i} className={`chat-bubble chat-bubble--${m.role}`}>
            {m.content}
          </div>
        ))}
        {isLoading && <div className="skeleton chat-bubble--ai" style={{ width: '60%', height: 40 }}></div>}
      </div>
      <div className="chat-input">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && sendMessage()}
          placeholder="Ask anything..."
          className="field__input"
        />
        <button onClick={sendMessage} className="btn btn--primary btn--sm">Send</button>
      </div>
    </div>
  );
}
```

### Rate Limit Handler

```javascript
/**
 * Rate limit handler with countdown timer
 * Shows banner, disables input, fires toast
 */
class RateLimitHandler {
  constructor(options = {}) {
    this.defaultWait = options.defaultWait || 60; // seconds
    this.onLimitStart = options.onLimitStart || (() => {});
    this.onLimitEnd = options.onLimitEnd || (() => {});
    
    this.isLimited = false;
    this.countdown = 0;
    this.timer = null;
  }

  /**
   * Handle a 429 response
   * @param {Response} response - Fetch response object
   */
  handleRateLimit(response) {
    // Get retry-after header (in seconds) or use default
    const retryAfter = parseInt(response.headers.get('Retry-After')) || this.defaultWait;
    
    this.startCountdown(retryAfter);
  }

  startCountdown(seconds) {
    if (this.isLimited) return; // Already in cooldown
    
    this.isLimited = true;
    this.countdown = seconds;
    
    // Show banner
    this.showBanner();
    
    // Disable send button
    this.disableSendButton();
    
    // Show toast
    this.showToast();
    
    // Callback
    this.onLimitStart(seconds);
    
    // Start timer
    this.timer = setInterval(() => {
      this.countdown--;
      this.updateUI();
      
      if (this.countdown <= 0) {
        this.endCountdown();
      }
    }, 1000);
  }

  endCountdown() {
    clearInterval(this.timer);
    this.isLimited = false;
    this.countdown = 0;
    
    // Hide banner
    this.hideBanner();
    
    // Enable send button
    this.enableSendButton();
    
    // Success toast
    swal.toast('You can send messages again', 'success');
    
    // Callback
    this.onLimitEnd();
  }

  showBanner() {
    const banner = document.getElementById('rate-limit-banner');
    if (banner) {
      banner.hidden = false;
      this.updateUI();
    }
  }

  hideBanner() {
    const banner = document.getElementById('rate-limit-banner');
    if (banner) {
      banner.hidden = true;
    }
  }

  updateUI() {
    // Update countdown text
    const countdownEl = document.getElementById('rate-limit-countdown');
    if (countdownEl) {
      countdownEl.textContent = this.countdown;
    }
    
    // Update progress bar
    const barEl = document.getElementById('rate-limit-bar');
    if (barEl && this.countdown > 0) {
      const totalDuration = parseInt(barEl.dataset.total) || this.defaultWait;
      const progress = (this.countdown / totalDuration) * 100;
      barEl.style.width = `${progress}%`;
    }
    
    // Update button countdown
    const sendBtn = document.querySelector('.chat-input .btn--primary');
    if (sendBtn) {
      sendBtn.dataset.countdown = this.countdown > 0 ? `(${this.countdown}s)` : '';
    }
  }

  disableSendButton() {
    const sendBtn = document.querySelector('.chat-input .btn--primary');
    if (sendBtn) {
      sendBtn.disabled = true;
      sendBtn.classList.add('rate-limited');
      sendBtn.setAttribute('aria-disabled', 'true');
    }
    
    // Also disable input
    const input = document.querySelector('.chat-input input, .chat-input textarea');
    if (input) {
      input.disabled = true;
    }
  }

  enableSendButton() {
    const sendBtn = document.querySelector('.chat-input .btn--primary');
    if (sendBtn) {
      sendBtn.disabled = false;
      sendBtn.classList.remove('rate-limited');
      sendBtn.removeAttribute('aria-disabled');
      sendBtn.dataset.countdown = '';
    }
    
    // Re-enable input
    const input = document.querySelector('.chat-input input, .chat-input textarea');
    if (input) {
      input.disabled = false;
      input.focus();
    }
  }

  showToast() {
    // Use swal.toast() helper from uiux-components.md
    // Note: If custom options needed beyond the helper, use Swal.fire() directly as fallback
    swal.toast('Slow down!', `Too many requests. Please wait ${this.countdown} seconds.`);
  }

  /**
   * Check if currently rate limited
   */
  get limited() {
    return this.isLimited;
  }

  /**
   * Get remaining seconds
   */
  get remaining() {
    return this.countdown;
  }
}

// Initialize rate limiter
const rateLimiter = new RateLimitHandler({
  defaultWait: 60,
  onLimitStart: (seconds) => console.log(`Rate limited for ${seconds}s`),
  onLimitEnd: () => console.log('Rate limit lifted'),
});
```

async function sendAIMessage(message) {
  // Check if rate limited
  if (rateLimiter.limited) {
    swal.toast(`Please wait ${rateLimiter.remaining} seconds`, 'warning');
    return;
  }

  const sendBtn = document.querySelector('.chat-input .btn--primary');
  sendBtn.classList.add('btn--loading');

  try {
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message }),
    });

    // Handle rate limit
    if (response.status === 429) {
      rateLimiter.handleRateLimit(response);
      return;
    }

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }

    const data = await response.json();
    displayAIResponse(data);
    
  } catch (error) {
    swal.error('Failed to send message', error.message);
  } finally {
    sendBtn.classList.remove('btn--loading');
  }
}

### React Hook Version

```jsx
import { useState, useCallback, useEffect, useRef } from 'react';

export function useRateLimit(defaultWait = 60) {
  const [isLimited, setIsLimited] = useState(false);
  const [countdown, setCountdown] = useState(0);
  const timerRef = useRef(null);

  const startLimit = useCallback((seconds = defaultWait) => {
    setIsLimited(true);
    setCountdown(seconds);

    // Show toast using swal.toast() helper from uiux-components.md
    // Note: If custom options needed beyond the helper, use Swal.fire() directly as fallback
    swal.toast('Rate limit reached', `Please wait ${seconds} seconds before trying again.`);
  }, [defaultWait]);

  useEffect(() => {
    if (countdown > 0) {
      timerRef.current = setInterval(() => {
        setCountdown((prev) => {
          if (prev <= 1) {
            setIsLimited(false);
            clearInterval(timerRef.current);
            return 0;
          }
          return prev - 1;
        });
      }, 1000);
    }

    return () => clearInterval(timerRef.current);
  }, [countdown > 0]); // Only restart when transitioning from 0 to positive

  const handleResponse = useCallback((response) => {
    if (response.status === 429) {
      const retryAfter = parseInt(response.headers.get('Retry-After')) || defaultWait;
      startLimit(retryAfter);
      return true; // Was rate limited
    }
    return false; // Not rate limited
  }, [startLimit, defaultWait]);

  return {
    isLimited,
    countdown,
    handleResponse,
    startLimit,
  };
}

// Usage in component
function ChatInput() {
  const { isLimited, countdown, handleResponse } = useRateLimit(60);
  const [message, setMessage] = useState('');

  const handleSend = async () => {
    if (isLimited) return;

    const response = await fetch('/api/chat', {
      method: 'POST',
      body: JSON.stringify({ message }),
    });

    if (handleResponse(response)) {
      return; // Rate limited, stop here
    }

    // Process response...
  };

  return (
    <div className="chat-input">
      <input
        value={message}
        onChange={(e) => setMessage(e.target.value)}
        disabled={isLimited}
        placeholder={isLimited ? `Wait ${countdown}s...` : 'Type a message...'}
      />
      <button
        onClick={handleSend}
        disabled={isLimited || !message.trim()}
        className={isLimited ? 'btn--primary rate-limited' : 'btn--primary'}
      >
        {isLimited ? `Wait (${countdown}s)` : 'Send'}
      </button>
    </div>
  );
}
```
